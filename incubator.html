import { createContext, useContext, useEffect, useState } from "react"; const ThemeContext = createContext(); export const ThemeProvider = ({ children }) => { const [theme, setTheme] = useState("dark"); useEffect(() => { document.documentElement.classList.remove("light","dark"); document.documentElement.classList.add(theme); }, [theme]); const toggleTheme = () => { setTheme(theme === "dark" ? "light" : "dark"); }; return ( <ThemeContext.Provider value={{ theme, toggleTheme }}> {children} </ThemeContext.Provider> ); }; export const useTheme = () => useContext(ThemeContext);
